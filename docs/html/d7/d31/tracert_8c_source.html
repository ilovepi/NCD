37 uint16_t ip_checksum(void* vdata, size_t length)
38 {
39  // Cast the data pointer to one that can be indexed.
40  char* data = (char*) vdata;
41 
42  // Initialize the accumulator.
43  uint64_t acc = 0xffff;
44 
45  // Handle any partial block at the start of the data.
46  unsigned int offset = ((uintptr_t) data) & 3;
47  if(offset){
48  size_t count = 4 - offset;
49  if(count > length)
50  count = length;
51  uint32_t word = 0;
52  memcpy(offset + (char*) &word, data, count);
53  acc += ntohl(word);
54  data += count;
55  length -= count;
56  }
57 
58  // Handle any complete 32-bit blocks.
59  char* data_end = data + (length & ~3);
60  while(data != data_end){
61  uint32_t word;
62  memcpy(&word, data, 4);
63  acc += ntohl(word);
64  data += 4;
65  }
66  length &= 3;
67 
68  // Handle any partial block at the end of the data.
69  if(length){
70  uint32_t word = 0;
71  memcpy(&word, data, length);
72  acc += ntohl(word);
73  }
74 
75  // Handle deferred carries.
76  acc = (acc & 0xffffffff) + (acc >> 32);
77  while(acc >> 16){
78  acc = (acc & 0xffff) + (acc >> 16);
79  }
80 
81  // If the data began at an odd byte address
82  // then reverse the byte order to compensate.
83  if(offset & 1){
84  acc = ((acc & 0xff00) >> 8) | ((acc & 0x00ff) << 8);
85  }
86 
87  // Return the checksum in network byte order.
88  return htons(~acc);
89 }
