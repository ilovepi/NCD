560 uint16_t ip_checksum(void* vdata, size_t length)
561 {
562  // Cast the data pointer to one that can be indexed.
563  char* data = (char*) vdata;
564 
565  // Initialize the accumulator.
566  uint64_t acc = 0xffff;
567 
568  // Handle any partial block at the start of the data.
569  unsigned int offset = ((uintptr_t) data) & 3;
570  if(offset){
571  size_t count = 4 - offset;
572  if(count > length)
573  count = length;
574  uint32_t word = 0;
575  memcpy(offset + (char*) &word, data, count);
576  acc += ntohl(word);
577  data += count;
578  length -= count;
579  }
580 
581  // Handle any complete 32-bit blocks.
582  char* data_end = data + (length & ~3);
583  while(data != data_end){
584  uint32_t word;
585  memcpy(&word, data, 4);
586  acc += ntohl(word);
587  data += 4;
588  }
589  length &= 3;
590 
591  // Handle any partial block at the end of the data.
592  if(length){
593  uint32_t word = 0;
594  memcpy(&word, data, length);
595  acc += ntohl(word);
596  }
597 
598  // Handle deferred carries.
599  acc = (acc & 0xffffffff) + (acc >> 32);
600  while(acc >> 16){
601  acc = (acc & 0xffff) + (acc >> 16);
602  }
603 
604  // If the data began at an odd byte address
605  // then reverse the byte order to compensate.
606  if(offset & 1){
607  acc = ((acc & 0xff00) >> 8) | ((acc & 0x00ff) << 8);
608  }
609 
610  // Return the checksum in network byte order.
611  return htons(~acc);
612 }
