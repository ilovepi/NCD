333 uint16_t ip_checksum(void* vdata, size_t length)
334 {
335  // Cast the data pointer to one that can be indexed.
336  char* data = (char*) vdata;
337 
338  // Initialize the accumulator.
339  uint64_t acc = 0xffff;
340 
341  // Handle any partial block at the start of the data.
342  unsigned int offset = ((uintptr_t) data) & 3;
343  if(offset){
344  size_t count = 4 - offset;
345  if(count > length)
346  count = length;
347  uint32_t word = 0;
348  memcpy(offset + (char*) &word, data, count);
349  acc += ntohl(word);
350  data += count;
351  length -= count;
352  }
353 
354  // Handle any complete 32-bit blocks.
355  char* data_end = data + (length & ~3);
356  while(data != data_end){
357  uint32_t word;
358  memcpy(&word, data, 4);
359  acc += ntohl(word);
360  data += 4;
361  }
362  length &= 3;
363 
364  // Handle any partial block at the end of the data.
365  if(length){
366  uint32_t word = 0;
367  memcpy(&word, data, length);
368  acc += ntohl(word);
369  }
370 
371  // Handle deferred carries.
372  acc = (acc & 0xffffffff) + (acc >> 32);
373  while(acc >> 16){
374  acc = (acc & 0xffff) + (acc >> 16);
375  }
376 
377  // If the data began at an odd byte address
378  // then reverse the byte order to compensate.
379  if(offset & 1){
380  acc = ((acc & 0xff00) >> 8) | ((acc & 0x00ff) << 8);
381  }
382 
383  // Return the checksum in network byte order.
384  return htons(~acc);
385 }
