951 uint16_t ip_checksum(void* vdata, size_t length)
952 {
953  // Cast the data pointer to one that can be indexed.
954  char* data = (char*) vdata;
955 
956  // Initialize the accumulator.
957  uint64_t acc = 0xffff;
958 
959  // Handle any partial block at the start of the data.
960  unsigned int offset = ((uintptr_t) data) & 3;
961  if(offset){
962  size_t count = 4 - offset;
963  if(count > length)
964  count = length;
965  uint32_t word = 0;
966  memcpy(offset + (char*) &word, data, count);
967  acc += ntohl(word);
968  data += count;
969  length -= count;
970  }
971 
972  // Handle any complete 32-bit blocks.
973  char* data_end = data + (length & ~3);
974  while(data != data_end){
975  uint32_t word = 0;
976  memcpy(&word, data, 4);
977  acc += ntohl(word);
978  data += 4;
979  }
980  length &= 3;
981 
982  // Handle any partial block at the end of the data.
983  if(length){
984  uint32_t word = 0;
985  memcpy(&word, data, length);
986  acc += ntohl(word);
987  }
988 
989  // Handle deferred carries.
990  acc = (acc & 0xffffffff) + (acc >> 32);
991  while(acc >> 16){
992  acc = (acc & 0xffff) + (acc >> 16);
993  }
994 
995  // If the data began at an odd byte address
996  // then reverse the byte order to compensate.
997  if(offset & 1){
998  acc = ((acc & 0xff00) >> 8) | ((acc & 0x00ff) << 8);
999  }
1000 
1001  // Return the checksum in network byte order.
1002  return htons(~acc);
1003 }
