846 u_int16_t ip_checksum(void* vdata, size_t length)
847 {
848  /* Cast the data pointer to one that can be indexed. */
849  char* data = (char*)vdata;
850 
851  /* Initialize the accumulator. */
852  u_int64_t acc = 0xffff;
853 
854  /* Handle any partial block at the start of the data. */
855  unsigned int offset = ((uintptr_t)data) & 3;
856  if(offset){
857  size_t count = 4 - offset;
858  if(count > length)
859  count = length;
860  u_int32_t word = 0;
861  memcpy(offset + (char*)&word, data, count);
862  acc += ntohl(word);
863  data += count;
864  length -= count;
865  }
866 
867  /* Handle any complete 32-bit blocks. */
868  char* data_end = data + (length & ~3);
869  while(data != data_end){
870  u_int32_t word = 0;
871  memcpy(&word, data, 4);
872  acc += ntohl(word);
873  data += 4;
874  }
875  length &= 3;
876 
877  /* Handle any partial block at the end of the data. */
878  if(length){
879  u_int32_t word = 0;
880  memcpy(&word, data, length);
881  acc += ntohl(word);
882  }
883 
884  /* Handle deferred carries. */
885  acc = (acc & 0xffffffff) + (acc >> 32);
886  while(acc >> 16){
887  acc = (acc & 0xffff) + (acc >> 16);
888  }
889 
890  /* If the data began at an odd byte address */
891  /* then reverse the byte order to compensate. */
892  if(offset & 1){
893  acc = ((acc & 0xff00) >> 8) | ((acc & 0x00ff) << 8);
894  }
895 
896  /* Return the checksum in network byte order. */
897  return htons(~acc);
898 }
